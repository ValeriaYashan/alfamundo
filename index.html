// AlfaMundo Apps Script v2 // Pegá este código completo en Extensiones → Apps Script function doGet(e) { return buildResponse(handleAction(e.parameter.action, null)); } function doPost(e) { let data = {}; try { data = JSON.parse(e.postData.contents); } catch(err) {} return buildResponse(handleAction(data.action, data)); } function buildResponse(result) { const out = ContentService.createTextOutput(JSON.stringify(result)); out.setMimeType(ContentService.MimeType.JSON); return out; } function handleAction(action, data) { try { if (action === 'getConfig') return getConfig(); if (action === 'saveConfig') return saveConfig(data.config); if (action === 'savePedido') return savePedido(data.pedido); return { error: 'Accion no reconocida: ' + action }; } catch(err) { return { error: err.toString() }; } } function getConfig() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const semana = getData(ss,'semana'); const productos= getData(ss,'productos'); const packaging= getData(ss,'packaging'); const delivery = getData(ss,'delivery'); const fechas = getData(ss,'fechas'); const cc = getData(ss,'Como comprar'); const qs = getData(ss,'Quienes somos'); // semana: fila 4 tiene los datos (fila 1=titulo, 2=nota, 3=headers, 4=datos) const semRow = semana.length > 3 ? semana[3] : (semana.length > 1 ? semana[1] : []); // productos: fila 4+ son datos (1=titulo,2=nota,3=headers,4+=datos) const prodStart = productos.findIndex(r => r[0] && r[1] && ['masivo','premium','premiado','saludable'].includes(String(r[1]).toLowerCase())); const prodRows = prodStart >= 0 ? productos.slice(prodStart) : []; // packaging: buscar filas con precio const packStart = packaging.findIndex(r => r[0] && r[1] !== undefined && !isNaN(Number(r[2])) && r[0] !== 'name' && !String(r[0]).includes('Config')); const packRows = packStart >= 0 ? packaging.slice(packStart) : []; // delivery: buscar filas con precio const delStart = delivery.findIndex(r => r[0] && !isNaN(Number(r[1])) && r[0] !== 'name' && !String(r[0]).includes('Config')); const delRows = delStart >= 0 ? delivery.slice(delStart) : []; // fechas: buscar filas con fechas reales const fechaRows = fechas.filter(r => r[0] && r[0] !== 'fecha' && !String(r[0]).includes('Config') && !String(r[0]).includes('semana') && !String(r[0]).includes('Cada')); // como comprar: buscar pasos const ccRows = cc.filter(r => r[0] && r[1] && !String(r[0]).includes('Config') && !String(r[0]).includes('Campo') && !String(r[1]).includes('Editá')); // quienes somos: buscar campos const qsRows = qs.filter(r => r[0] && r[1] && !String(r[0]).includes('Config') && !String(r[0]).includes('Campo') && !String(r[1]).includes('Editá')); return { ok: true, week: { name: String(semRow[0]||''), theme: String(semRow[1]||''), contenido: String(semRow[2]||''), date: String(semRow[3]||''), badge: String(semRow[4]||'') }, products: prodRows.filter(r=>r[0]).map((r,i)=>({ id:'p'+i, name: String(r[0]||''), cat: String(r[1]||'masivo'), desc: String(r[2]||''), price: Number(r[3])||0, img: String(r[4]||''), stock: String(r[5]||'Stock').toLowerCase().trim() === 'agotado' ? 'agotado' : 'stock' })), packaging: packRows.filter(r=>r[0]).map((r,i)=>({ id:'pk'+i, name:String(r[0]||''), desc:String(r[1]||''), price:Number(r[2])||0 })), delivery: delRows.filter(r=>r[0]).map((r,i)=>({ id:'dl'+i, name:String(r[0]||''), price:Number(r[1])||0 })), fechas: fechaRows.map(r=>{ const v = r[0]; if (v instanceof Date) { const d=v.getDate().toString().padStart(2,'0'); const m=v.getMonth()+1; const meses=['enero','febrero','marzo','abril','mayo','junio','julio','agosto','septiembre','octubre','noviembre','diciembre']; return `${d} de ${meses[m-1]}`; } return String(v); }), comoComprar: ccRows.map(r=>({ paso:String(r[0]||''), desc:String(r[1]||'') })), quienesSomos: Object.fromEntries(qsRows.map(r=>[String(r[0]).trim(), String(r[1]||'').trim()])) }; } function saveConfig(config) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const shS = ss.getSheetByName('semana'); // Buscar la fila de datos (fila 4 en el nuevo formato) if (shS.getLastRow() < 4) { shS.appendRow(['','','','','']); } shS.getRange(4,1,1,5).setValues([[ config.week.name, config.week.theme, config.week.contenido||'', config.week.date, config.week.badge ]]); // Productos: encontrar donde empiezan los datos const shP = ss.getSheetByName('productos'); const lastP = shP.getLastRow(); if (lastP >= 4) shP.deleteRows(4, lastP-3); if (config.products && config.products.length > 0) { shP.getRange(4,1,config.products.length,6).setValues( config.products.map(p=>[p.name,p.cat,p.desc,p.price,p.img||'',p.stock==='agotado'?'Agotado':'Stock']) ); } writeSheet(ss,'packaging',3, (config.packaging||[]).map(p=>[p.name,p.desc,p.price]),4); writeSheet(ss,'delivery',2, (config.delivery||[]).map(d=>[d.name,d.price]),4); writeSheet(ss,'fechas',1, (config.fechas||[]).map(f=>[f]),4); return { ok: true }; } function savePedido(pedido) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName('pedidos'); sh.appendRow([ pedido.fecha||'', pedido.hora||'', pedido.nombre||'', pedido.whatsapp||'', pedido.email||'', pedido.productos||'', pedido.unidades||'', pedido.packaging||'', pedido.envio||'', pedido.total||'', pedido.fechaEntrega||'', pedido.ocasion||'', pedido.restricciones||'', pedido.msgTarjeta||'', 'Pendiente' ]); return { ok: true }; } function getData(ss, name) { const sh = ss.getSheetByName(name); if (!sh || sh.getLastRow() < 1) return [[]]; return sh.getRange(1,1,sh.getLastRow(),Math.max(sh.getLastColumn(),1)).getValues(); } function writeSheet(ss, name, cols, rows, startRow) { const sh = ss.getSheetByName(name); const sr = startRow || 2; const last = sh.getLastRow(); if (last >= sr) sh.deleteRows(sr, last-sr+1); if (rows.length > 0) sh.getRange(sr,1,rows.length,cols).setValues(rows); }
